107 /*
108  * Transfers a valid ASCII string describing a CAN frame into struct canfd_frame.
109  *
110  * CAN 2.0 frames (aka Classical CAN)
111  * - string layout <can_id>#{R{len}|data}{_len8_dlc}
112  * - {data} has 0 to 8 hex-values that can (optionally) be separated by '.'
113  * - {len} can take values from 0 to 8 and can be omitted if zero
114  * - {_len8_dlc} can take hex values from '_9' to '_F' when len is CAN_MAX_DLEN
115  * - return value on successful parsing: CAN_MTU
116  *
117  * CAN FD frames
118  * - string layout <can_id>##<flags>{data}
119  * - <flags> a single ASCII Hex value (0 .. F) which defines canfd_frame.flags
120  * - {data} has 0 to 64 hex-values that can (optionally) be separated by '.'
121  * - return value on successful parsing: CANFD_MTU
122  *
123  * Return value on detected problems: 0
124  *
125  * <can_id> can have 3 (standard frame format) or 8 (extended frame format)
126  * hexadecimal chars
127  *
128  *
129  * Examples:
130  *
131  * 123# -> standard CAN-Id = 0x123, len = 0
132  * 12345678# -> extended CAN-Id = 0x12345678, len = 0
133  * 123#R -> standard CAN-Id = 0x123, len = 0, RTR-frame
134  * 123#R0 -> standard CAN-Id = 0x123, len = 0, RTR-frame
135  * 123#R7 -> standard CAN-Id = 0x123, len = 7, RTR-frame
136  * 123#R8_9 -> standard CAN-Id = 0x123, len = 8, dlc = 9, RTR-frame
137  * 7A1#r -> standard CAN-Id = 0x7A1, len = 0, RTR-frame
138  *
139  * 123#00 -> standard CAN-Id = 0x123, len = 1, data[0] = 0x00
140  * 123#1122334455667788 -> standard CAN-Id = 0x123, len = 8
141  * 123#1122334455667788_E -> standard CAN-Id = 0x123, len = 8, dlc = 14
142  * 123#11.22.33.44.55.66.77.88 -> standard CAN-Id = 0x123, len = 8
143  * 123#11.2233.44556677.88 -> standard CAN-Id = 0x123, len = 8
144  * 32345678#112233 -> error frame with CAN_ERR_FLAG (0x2000000) set
145  *
146  * 123##0112233 -> CAN FD frame standard CAN-Id = 0x123, flags = 0, len = 3
147  * 123##1112233 -> CAN FD frame, flags = CANFD_BRS, len = 3
148  * 123##2112233 -> CAN FD frame, flags = CANFD_ESI, len = 3
149  * 123##3 -> CAN FD frame, flags = (CANFD_ESI | CANFD_BRS), len = 0
150  * ^^
151  * CAN FD extension to handle the canfd_frame.flags content
152  *
153  * Simple facts on this compact ASCII CAN frame representation:
154  *
155  * - 3 digits: standard frame format
156  * - 8 digits: extendend frame format OR error frame
157  * - 8 digits with CAN_ERR_FLAG (0x2000000) set: error frame
158  * - an error frame is never a RTR frame
159  * - CAN FD frames do not have a RTR bit
160  */

164 /*
165  * Creates a CAN frame hexadecimal output in compact format.
166  * The CAN data[] is separated by '.' when sep != 0.
167  *
168  * The type of the CAN frame (CAN 2.0 / CAN FD) is specified by maxdlen:
169  * maxdlen = 8 -> CAN2.0 frame (aka Classical CAN)
170  * maxdlen = 64 -> CAN FD frame
171  *
172  * 12345678#112233 -> extended CAN-Id = 0x12345678, len = 3, data, sep = 0
173  * 123#1122334455667788_E -> standard CAN-Id = 0x123, len = 8, dlc = 14, data, sep = 0
174  * 12345678#R -> extended CAN-Id = 0x12345678, RTR, len = 0
175  * 12345678#R5 -> extended CAN-Id = 0x12345678, RTR, len = 5
176  * 123#11.22.33.44.55.66.77.88 -> standard CAN-Id = 0x123, dlc = 8, sep = 1
177  * 32345678#112233 -> error frame with CAN_ERR_FLAG (0x2000000) set
178  * 123##0112233 -> CAN FD frame standard CAN-Id = 0x123, flags = 0, len = 3
179  * 123##2112233 -> CAN FD frame, flags = CANFD_ESI, len = 3
180  *
181  * Examples:
182  *
183  * fprint_canframe(stdout, &frame, "\n", 0); // with eol to STDOUT
184  * fprint_canframe(stderr, &frame, NULL, 0); // no eol to STDERR
185  *
186  */
